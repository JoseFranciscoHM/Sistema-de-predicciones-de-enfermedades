\section{Introducción}
% =============================================================================

\begin{frame}{Motivación}
  \begin{itemize}
    \item Las enfermedades infecciosas representan un desafío constante para la salud pública
    \item San Luis Potosí presenta incidencia de hantavirus, COVID-19 y dengue
    \item Las herramientas predictivas permiten anticipar brotes y optimizar recursos
    \item Integración de datos epidemiológicos + tendencias de búsqueda web (Google Trends)
  \end{itemize}
  \vspace{0.5cm}
  \begin{alertblock}{Objetivo}
    Construir un sistema de predicción temprana basado en RNN que integre
    datos epidemiológicos y tendencias de búsqueda para 58 municipios de SLP.
  \end{alertblock}
\end{frame}

\begin{frame}{Arquitectura del Sistema}
  \centering
  % \includegraphics[width=0.9\textwidth]{diagrama_arquitectura.png}
  
  \begin{enumerate}
    \item Recolección: Google Trends + APIs gubernamentales + CSV manual
    \item Almacenamiento: SQLite con 5 tablas normalizadas
    \item Modelo: LSTM bidireccional (2 capas, 64--128 unidades)
    \item Visualización: Dashboard web con Plotly.js
  \end{enumerate}
\end{frame}

\section{Fundamentos Teóricos}
% =============================================================================

\begin{frame}{Redes Neuronales Recurrentes (RNN)}
  \begin{itemize}
    \item Las RNN capturan dependencias temporales en secuencias
    \item A diferencia de las CNN, mantienen un estado oculto que evoluciona con el tiempo
  \end{itemize}

  \begin{equation*}
    h_t = \tanh(W_{ih} x_t + b_{ih} + W_{hh} h_{t-1} + b_{hh})
  \end{equation*}

  \begin{itemize}
    \item \textbf{Problema}: gradientes explosivos/desvanecientes en secuencias largas
    \item \textbf{Solución}: LSTM (Long Short-Term Memory) con compuertas de olvido, entrada y salida
  \end{itemize}
\end{frame}

\begin{frame}{LSTM — Long Short-Term Memory}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Compuertas:}
    \begin{itemize}
      \item Olvido: $f_t = \sigma(W_f \cdot [h_{t-1}, x_t] + b_f)$
      \item Entrada: $i_t = \sigma(W_i \cdot [h_{t-1}, x_t] + b_i)$
      \item Salida: $o_t = \sigma(W_o \cdot [h_{t-1}, x_t] + b_o)$
    \end{itemize}
    
    \column{0.48\textwidth}
    \textbf{Estado de celda:}
    \begin{itemize}
      \item $\tilde{C}_t = \tanh(W_C \cdot [h_{t-1}, x_t] + b_C)$
      \item $C_t = f_t * C_{t-1} + i_t * \tilde{C}_t$
      \item $h_t = o_t * \tanh(C_t)$
    \end{itemize}
  \end{columns}
  \vspace{0.3cm}
  \begin{itemize}
    \item La capa bidireccional procesa la secuencia en ambas direcciones
    \item Captura patrones antes y después de cada punto temporal
  \end{itemize}
\end{frame}

\begin{frame}{Arquitectura del Modelo}
  \centering
  \begin{tabular}{lc}
    \toprule
    \textbf{Componente} & \textbf{Detalle} \\
    \midrule
    Entrada & $21$ días $\times$ $N$ características \\
    Capa LSTM Bidireccional 1 & 64 unidades, return sequences \\
    Dropout & 0.3 \\
    Capa LSTM Bidireccional 2 & 128 unidades \\
    Dropout & 0.3 \\
    Salida 1 (Clasificación) & Sigmoide $\rightarrow$ probabilidad de brote \\
    Salida 2 (Regresión) & ReLU $\rightarrow$ casos estimados en 7 días \\
    \bottomrule
  \end{tabular}

  \begin{itemize}
    \item Pérdida combinada: binary\_crossentropy + mse
    \item Optimizador: Adam, learning rate $10^{-3}$
    \item Early stopping con paciencia de 10 épocas
  \end{itemize}
\end{frame}

\section{Metodología}
% =============================================================================

\begin{frame}{Pipeline de Datos}
  \begin{enumerate}
    \item \textbf{Google Trends}: Extracción por palabra clave por municipio (vía pytrends)
      \begin{itemize}
        \item 5--7 palabras clave por enfermedad
        \item Fallback graceful ante HTTP 429
      \end{itemize}
    \item \textbf{Datos epidemiológicos}: Fuentes oficiales (Datos Abiertos México, SINAIS)
      \begin{itemize}
        \item Casos confirmados, hospitalizaciones, muertes
        \item Stubs para desarrollo sin conexión real
      \end{itemize}
    \item \textbf{Importación CSV}: Alternativa manual con validación de formato
    \item \textbf{Datos sintéticos}: Generación con patrones estacionales para 58 municipios
  \end{enumerate}
\end{frame}

\begin{frame}{Características por Enfermedad}
  Cada enfermedad tiene un conjunto específico de palabras clave de Google Trends:
  
  \begin{table}
    \centering
    \begin{tabular}{lcc}
      \toprule
      \textbf{Enfermedad} & \textbf{Keywords} & \textbf{Características} \\
      \midrule
      Dengue & 8 & 8 + 3 = 11 \\
      Hantavirus & 10 & 10 + 3 = 13 \\
      COVID-19 & 7 & 7 + 3 = 10 \\
      \bottomrule
    \end{tabular}
  \end{table}
  
  Las 3 características epidemiológicas comunes:
  \begin{itemize}
    \item Casos confirmados (normalizados $\div 100$)
    \item Hospitalizaciones (normalizadas $\div 20$)
    \item Muertes (normalizadas $\div 5$)
  \end{itemize}
\end{frame}

\begin{frame}{Generación de Datos Sintéticos}
  \begin{itemize}
    \item Período: 2 años (730 días)
    \item 58 municipios con pesos poblacionales
    \item Patrones estacionales realistas:
      \begin{itemize}
        \item \textbf{Dengue}: pico en temporada de lluvias (junio--octubre)
        \item \textbf{COVID-19}: pico en invierno (diciembre--febrero)
        \item \textbf{Hantavirus}: brotes en primavera y otoño
      \end{itemize}
    \item Ruido aleatorio para simular variabilidad real
    \item 29,750 registros de tendencias + 152,250 registros epidemiológicos
  \end{itemize}
\end{frame}
